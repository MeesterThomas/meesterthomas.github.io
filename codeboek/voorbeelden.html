        <meta charset="utf-8">
        <link rel="stylesheet" href="../css/index.css">
    </head>
    <body>
        <p>Deze tag wordt alleen via de tag opgemaakt.</p>
        <p class="tekstClass">Deze tag wordt via de tag en de class opgemaakt.</p>

Tekst veranderen met een knop


HTML


<!DOCTYPE html>
<html lang="en">
    <head>
        <title>Titel van je website</title>
        <meta charset="utf-8">
    </head>
    <body>
        <button onclick="DrukKnop()">Verander!</button>
        <p id="veranderTekst">Dit is normale tekst.</p>
    </body>
</html>

CSS

Javascript


function DrukKnop()
{
    document.getElementById("veranderTekst").innerHTML = "De tekst is veranderd!";
    document.getElementById("veranderTekst").style.color = "rgb(11, 11, 177)";
}

CSS

Javascript


let teller = 0;

function ZingKnop()
{
    teller += 1;
    if (teller >= 5)
        return; //De function stopt hier als het lied al voorbij is.
    document.getElementById("liedTekst0").innerHTML = KrijgLiedTekst();
}

function KrijgLiedTekst() //Deze function geeft tekst terug bij return.
{
    if (teller == 1 || teller == 2 || teller = 4)
        return "Hoofd, schouders, knie en teen, knie en teen!";
    else if (teller == 3)
        return "Oren, ogen, puntje van je neus!";
    else
        return "";
}
